Ejemplo: Almacén de Datos 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.18;

contract MiAlmacen {
    string private mensaje;
    address public owner;

    event MensajeCambiado(string nuevoMensaje);

    constructor() {
        owner = msg.sender;
    }

    // Cambiar datos (Cuesta Gas)
    function guardar(string memory _nuevo) public {
        mensaje = _nuevo;
        emit MensajeCambiado(_nuevo);
    }

    // Leer datos (Gratis)
    function leer() public view returns (string memory) {
        return mensaje;
    }
}

Anatomía de un Contrato Inteligente

1. El Estado (Mappings)

En lugar de una base de datos SQL, usamos mappings. Es como un diccionario gigante de clave-valor que vive en la blockchain.

// Clave: Dirección (0x...) -> Valor: Saldo (100)
mapping(address => uint256) public saldos;

function recargar() public payable {
    saldos[msg.sender] += msg.value;
}

2. El Control (Modifiers)

Los modifiers son como guardias de seguridad. Se ejecutan antes que la función para validar quién puede entrar.

modifier soloOwner() {
    require(msg.sender == owner, "No eres el dueño");
    _; // El guion bajo indica "continúa con la función"
}

function retirar() public soloOwner {
    payable(owner).transfer(address(this).balance);
}

3. La Comunicación (Events)

La blockchain no puede "avisar" a tu web que algo pasó. Para eso usamos Events. Tu frontend escucha estos eventos para actualizar la UI en tiempo real.

// En Solidity:
event PagoRecibido(address pagador, uint256 monto);

function pagar() public payable {
    emit PagoRecibido(msg.sender, msg.value);
}
// En JavaScript (Ethers.js):
contract.on("PagoRecibido", (pagador, monto) => {
    console.log(`¡Pago de ${pagador} por ${monto}!`);
    actualizarUI();
});

La EVM: El Corazón del Sistema

La **Ethereum Virtual Machine (EVM)** es un computador global descentralizado. Tu código Solidity se compila en un lenguaje de bajo nivel llamado Opcodes (ej: PUSH1, MSTORE, CALL).

Cada operación consume una cantidad específica de Gas. Si tu código es ineficiente, tus usuarios pagarán más por interactuar con él.

GAS OPTIMIZATION TIPS
  • • Usa external en vez de public si solo se llama desde fuera.
  • • Evita loops (bucles) sobre arrays que pueden crecer infinitamente.
  • • Guarda solo lo estrictamente necesario en el estado.

Gas y Eficiencia

Cada línea de código en el backend cuesta dinero. Aprenderás a usar uint8 en lugar de uint256 para ahorrarle dinero a tus usuarios.

Seguridad (Reentrancy)

A diferencia de la Web2, un bug aquí puede vaciar la billetera de todos. Veremos por qué el orden de las líneas importa.

Verificación

Aprende a publicar tu código en Etherscan para que el mundo vea que tu backend es honesto y transparente.
